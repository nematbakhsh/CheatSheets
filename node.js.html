Install from nodejs.org

NPM

node -v | node --version returns node.js version npm -v | npm --version returns npm version npm init initializes an app by creating a package.json file

packaje.json content

"name" : STRING "version" : STRING "description" : STRING "keywords" : Array array-of-strings "homepage" : STRING "bugs" : { "url" : STRING, "email" : STRING} "main" : STRING "scripts" : OBJECT "author" : STRING "license" : STRING "dependencies" : OBJECT "devDependencies" : OBJECT "repository" : { "type" : STRING, "url" : STRING "private" : true npm will refuse to publish private apps
