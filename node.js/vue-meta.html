Node.js Vue-Meta

Manage HTML metadata in Vue.js components with SSR support

Installation

npm install vue-meta --save

Boiler Plate

Inside main.js or router.js (better)
import Vue from 'vue'
import Router from 'vue-router'
import VueMeta from 'vue-meta'

Vue.use(Router)
Vue.use(VueMeta, {
  // optional pluginOptions
  keyName: 'metaInfo',
  refreshOnceOnNavigation: true
})
Inside Component.Vue
<template>
  <div id="app">
    <router-view></router-view>
  </div>
</template>
 
<script>
  export default {
    metaInfo: {
      // if no subcomponents specify a metaInfo.title, this title will be used
      title: 'My Example App',
      // all titles will be injected into this template
      titleTemplate: '%s - Yay!', // use null to override the parent template and just use the above title only
      htmlAttrs: {
        lang: 'en',
        amp: true
      },
      meta: [
        { name: "viewport", content: "width=device-width, initial-scale=1" },
        { name: "description", content: "website description" },
        { charset: "utf-8" },
        // The Open Graph protocol
        { property: "og:title", content: "website title" },
        { property: "og:site_name", content: "website name" },
        { property: "og:type", content: "website" },
        { property: "og:image", content: "image url"}
        { property: "og:url", content: "website url" },
        { property: "og:description", content: "website description" }
      ]
    }
  }
</script>
