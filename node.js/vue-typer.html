Node.js Vue-Typer

Installation

npm install --save vue-typer

Usage

In Component.vue
<template>
<vue-typer text='Hello World! I was registered locally!'></vue-typer>
</template>

import { VueTyper } from 'vue-typer'

var MyComponent = {
  // ...
  components: {
    // ES6; property shorthand + Vue should automatically dasherize the key for us
    VueTyper
    // pre-ES6
    'vue-typer': VueTyper
  }
}
